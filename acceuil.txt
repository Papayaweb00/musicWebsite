var playpartie = document.querySelector(".playpartie");

button.addEventListener('click', play);

function play () {
    if (music.paused) {
        var audio = new Audio('Musiques/Alan_Walker_&_Ava_Max_-_Alone,_Pt._II(360p)_20220911_133217.mp3');
        audio.play();
        playpartie.innerHTML = "<i class='fa-solid fa-pause play'></i>";
    } else {
        audio.play();
        playpartie.innerHTML = "<i class='fa-solid fa-play play'></i>";
    }
}
